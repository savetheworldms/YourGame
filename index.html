<body>

    <h1>ЮРА 🐛</h1>

    <p>Собирай сыр-косичку (🧀) и избегай корюшку (🐟)!</p>

    <canvas id="gameCanvas" width="400" height="400"></canvas>


    <div id="gameOverScreen">

        <h2>ЮРА наелся корюшки и проиграл! 🐟💀</h2>

        <button id="restartBtn">Начать заново</button>

    </div>


    <script>

        let canvas, ctx, snake, direction, cheese, fish, gameOver;


        function init() {

            canvas = document.getElementById("gameCanvas");

            ctx = canvas.getContext("2d");


            document.getElementById("gameOverScreen").style.display = "none";


            cheese = randomPosition(); 

            fish = randomPosition(); 


            snake = [getSafePosition()];

            direction = { x: 0, y: 0 };

            gameOver = false;


            document.addEventListener("keydown", changeDirection);

            requestAnimationFrame(gameLoop);

        }


        function randomPosition() {

            const gridSize = 20;

            return {

                x: Math.floor(Math.random() * (canvas.width / gridSize)) * gridSize,

                y: Math.floor(Math.random() * (canvas.height / gridSize)) * gridSize

            };

        }


        function getSafePosition() {

            let pos;

            do {

                pos = randomPosition();

            } while (fish && pos.x === fish.x && pos.y === fish.y);

            return pos;

        }


        function draw() {

            if (gameOver) return;


            ctx.clearRect(0, 0, canvas.width, canvas.height);


            ctx.fillStyle = "lime";

            snake.forEach((segment, index) => {

                ctx.fillRect(segment.x, segment.y, 20, 20);

                if (index === 0) {

                    ctx.fillStyle = "green";

                }

            });


            ctx.fillStyle = "yellow";

            ctx.fillRect(cheese.x, cheese.y, 20, 20);


            ctx.fillStyle = "red";

            ctx.fillRect(fish.x, fish.y, 20, 20);

        }


        function update() {

            if (gameOver) return;


            const newHead = { x: snake[0].x + direction.x * 20, y: snake[0].y + direction.y * 20 };


            if (newHead.x < 0 || newHead.x >= canvas.width || newHead.y < 0 || newHead.y >= canvas.height) {

                endGame();

                return;

            }


            if (snake.some(segment => segment.x === newHead.x && segment.y === newHead.y)) {

                endGame();

                return;

            }


            if (newHead.x === cheese.x && newHead.y === cheese.y) {

                cheese = randomPosition();

                fish = randomPosition();

            } else {

                snake.pop();

            }


            if (newHead.x === fish.x && newHead.y === fish.y) {

                endGame();

                return;

            }


            snake.unshift(newHead);

        }


        function endGame() {

            gameOver = true;

            document.getElementById("gameOverScreen").style.display = "block";

        }


        function changeDirection(event) {

            if (gameOver) return;

            const key = event.key;

            if (key === "ArrowUp" && direction.y === 0) {

                direction = { x: 0, y: -1 };

            } else if (key === "ArrowDown" && direction.y === 0) {

                direction = { x: 0, y: 1 };

            } else if (key === "ArrowLeft" && direction.x === 0) {

                direction = { x: -1, y: 0 };

            } else if (key === "ArrowRight" && direction.x === 0) {

                direction = { x: 1, y: 0 };

            }

        }


        function gameLoop() {

            if (!gameOver) {

                update();

                draw();

                requestAnimationFrame(gameLoop);

            }

        }


        document.getElementById("restartBtn").addEventListener("click", () => {

            init();

        });


        init();

    </script>
